flink 加速启动


# 上传flink安装包到hadoop(只有最初一次需要执行)
FLINK_DIR=/data/flink-1.14.5
# 远端hadoopFlink安装目录
HADOOP_FLINK_DIR=/flink/flink-1.14.5

## 清理之前的包,如果存在的话
hadoop fs -rm -r -f $HADOOP_FLINK_DIR
hadoop fs -rm -r -f .Trash
## 进入安装目录
cd $FLINK_DIR
hadoop fs -put . /flink/

# 当前打包模组
PROJECT_NAME={这里填项目名}
PROJECT_PATH={这里填项目路径}
CURRENT_MODULE={这里填项目模块名}
HADOOP_URL={这里填hadoop网址,例如: test01:8020}
JOB_NAME={这里填任务名称}

# 打包对应的服务(每次执行)
cd $PROJECT_PATH && ./gradlew clean generateRuntimeConfig -Denv=pre -Dcluster=docker --daemon -p $CURRENT_MODULE

# 创建module文件夹
hadoop fs -mkdir /flink/$CURRENT_MODULE

# 记录 sha256 值,这个需要存一下,以后用来清除无用的依赖列表
cd $PROJECT_PATH/$CURRENT_MODULE/build/$PROJECT_NAME && find $PROJECT_PATH/$CURRENT_MODULE/build/$PROJECT_NAME/lib/ -type f -print0 | sort -z | xargs -0 sha256sum | sha1sum | awk '{print $1}'> $CURRENT_MODULE.sha

# 设置一下环境变量
# HDFS_DEFAULT=$(xmllint --xpath "//property[name='fs.defaultFS']/value/text()" $HADOOP_HOME/etc/hadoop/conf.cloudera.yarn/core-site.xml)
HDFS_DEFAULT=$(hdfs getconf -confKey fs.defaultFS)
CURRENT_MODULE_TASK_SHA=$(cat $PROJECT_PATH/$CURRENT_MODULE/build/$PROJECT_NAME/$CURRENT_MODULE.sha)
flink_base_bin=/tmp/flink-1.14.5/bin

echo HDFS_DEFAULT:$HDFS_DEFAULT
echo CURRENT_MODULE_TASK_SHA:$CURRENT_MODULE_TASK_SHA
echo flink_base_bin:$flink_base_bin

# 检查依赖是否存在
hadoop fs -ls /flink/$CURRENT_MODULE/$CURRENT_MODULE_TASK_SHA
if [ $? -eq 0 ]; then
  echo "Directory exists."
else
    # 不存在则上传所有依赖(如果依赖发生了变动则需要执行,TODO 需要增加判断变动的脚本)
    cd $PROJECT_PATH/$CURRENT_MODULE/build/$PROJECT_NAME/lib && hadoop fs -mkdir /flink/$CURRENT_MODULE/$CURRENT_MODULE_TASK_SHA && hadoop fs -put . /flink/$CURRENT_MODULE/$CURRENT_MODULE_TASK_SHA
fi

# 启动对应服务
# $flink_base_bin/flink run -d -m yarn-cluster -p 1 -ys 1 -yjm 2048m \
# -yD yarn.provided.lib.dirs="$HDFS_DEFAULT$HADOOP_FLINK_DIR/lib;$HDFS_DEFAULT$HADOOP_FLINK_DIR/plugins;$HDFS_DEFAULT/flink/$CURRENT_MODULE/$CURRENT_MODULE_TASK_SHA/lib" \
# -yD env.java.opts="-DAPP_NAME=$PROJECT_NAME -DJOB_ENV=dev" -ytm 3072m -ynm $JOB_NAME \
# $PROJECT_PATH/$CURRENT_MODULE/build/libs/$PROJECT_NAME-$CURRENT_MODULE-1.0.0.jar {任务启动参数}


$flink_base_bin/flink run-application \
-t yarn-application \
# yjm
-Djobmanager.memory.process.size=2048m \
# ytm
-Dtaskmanager.memory.process.size=3072m \
# ys
-Dtaskmanager.numberOfTaskSlots=1 \
# p
-Dparallelism.default=1 \
# ynm
-Dyarn.application.name=$JOB_NAME \
-Dyarn.provided.usrlib.dir="$HDFS_DEFAULT/flink/$CURRENT_MODULE/$CURRENT_MODULE_TASK_SHA/lib" \
-Dyarn.provided.lib.dirs="$HDFS_DEFAULT$HADOOP_FLINK_DIR/lib;$HDFS_DEFAULT$HADOOP_FLINK_DIR/plugins;$HDFS_DEFAULT/flink/$CURRENT_MODULE/$CURRENT_MODULE_TASK_SHA/lib" \
-Denv.java.opts="-DAPP_NAME=$PROJECT_NAME -DJOB_ENV=dev" \
$HDFS_DEFAULT/flink/$PROJECT_NAME-$CURRENT_MODULE.jar {任务启动参数}



su - {这里填user_name} -c '$FLINK_DIRbin/flink run-application -t yarn-application \
